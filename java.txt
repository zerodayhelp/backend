// Read file lines
async function readFileLines(filePath) {
  try {
    const content = await fs.readFile(filePath, 'utf8');
    return content.split('\n')
      .map(line => line.trim())
      .filter(line => line && !line.startsWith('#'));
  } catch (error) {
    if (error.code === 'ENOENT') {
      return [];
    }
    throw error;
  }
}

// Parse proxy string
function parseProxy(proxyString) {
  try {
    // Format: ip:port:username:password or ip:port or protocol://user:pass@host:port
    if (proxyString.includes('@')) {
      // Already in URL format
      return new URL(proxyString);
    } else {
      // Parse ip:port:username:password format
      const parts = proxyString.split(':');
      if (parts.length === 4) {
        const [ip, port, username, password] = parts;
        return new URL(`http://${username}:${password}@${ip}:${port}`);
      } else if (parts.length === 2) {
        // ip:port format
        const [ip, port] = parts;
        return new URL(`http://${ip}:${port}`);
      }
    }
    return null;
  } catch (error) {
    log(`Error parsing proxy: ${proxyString} - ${error.message}`, 'WARN');
    return null;
  }
}

// Get random user agent
async function getRandomUserAgent() {
  try {
    if (CONFIG.rotateUserAgent) {
      // Read user agents from file
      const userAgents = await readFileLines(CONFIG.userAgentFile);
      if (userAgents.length > 0) {
        const randomIndex = Math.floor(Math.random() * userAgents.length);
        return userAgents[randomIndex];
      }
    }
  } catch (error) {
    log(`Error reading user agents file: ${error.message}`, 'WARN');
  }
  
  // Default fallback
  return 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/145.0.0.0 Safari/537.36';
}

// Get random proxy
async function getRandomProxy(proxyList) {
  // Check if we should use proxies for API calls
  if (!CONFIG.useProxyForApi || proxyList.length === 0) {
    return null;
  }
  
  const randomIndex = Math.floor(Math.random() * proxyList.length);
  return parseProxy(proxyList[randomIndex]);
}

// Initialize browser with proxy (ONLY for solving challenge, NO PROXY)
async function initBrowser(proxyUrl = null, userAgent = null) {
  const launchOptions = {
    headless: CONFIG.headlessMode,
    defaultViewport: { width: 1280, height: 800 },
    args: [
      '--no-sandbox',
      '--disable-setuid-sandbox',
      '--disable-web-security',
      '--disable-features=IsolateOrigins,site-per-process',
      '--disable-blink-features=AutomationControlled'
    ]
  };
  
  // IMPORTANT: DON'T add proxy for browser - use clean connection to solve Cloudflare
  if (CONFIG.useProxyForBrowser && proxyUrl) {
    log('WARNING: Using proxy for browser may cause Cloudflare issues!', 'WARN');
    const proxyHost = `${proxyUrl.hostname}:${proxyUrl.port}`;
    launchOptions.args.push(`--proxy-server=${proxyUrl.protocol}//${proxyHost}`);
    log(`Using proxy: ${proxyHost}`, 'PROXY');
  } else {
    log('Using clean connection (no proxy) for Cloudflare challenge', 'INFO');
  }
  
  // Add user agent if provided
  if (userAgent) {
    launchOptions.args.push(`--user-agent=${userAgent}`);
  }
  
  try {
    const browser = await puppeteer.launch(launchOptions);
    return { browser, proxyUrl };
  } catch (error) {
    log(`Failed to launch browser: ${error.message}`, 'ERROR');
    throw error;
  }
}

// Solve challenge and get session
async function solveChallenge(page, email, retryCount, proxyUrl) {
  if (retryCount === undefined) retryCount = 0;
  if (proxyUrl === undefined) proxyUrl = null;

  var maxRetries = CONFIG.maxRetries;

  // Pre-define all URL/string constants as local vars so obfuscator
  // renames them consistently in Node scope — never serialized into browser
  var URL_HOME        = 'https://react.email';
  var URL_MARKDOWN    = 'https://react.email/components/markdown';
  var STR_CHECKPOINT  = 'Security Checkpoint';
  var WAIT_IDLE       = 'networkidle2';

  try {
    log('Solving challenge for ' + email + ' (attempt ' + (retryCount + 1) + '/' + maxRetries + ')...');

    await page.goto(URL_HOME, { waitUntil: WAIT_IDLE, timeout: 30000 });
    await new Promise(function(resolve) { setTimeout(resolve, 3000); });

    var pageTitle = await page.title();
    if (pageTitle.indexOf(STR_CHECKPOINT) !== -1) {
      log('Security challenge detected, waiting for completion...', 'CHALLENGE');

      // Pass the checkpoint string as an argument so it's never inside the serialized fn body
      await page.waitForFunction(
        function(chk) { return document.title.indexOf(chk) === -1; },
        { timeout: 30000 },
        STR_CHECKPOINT
      ).catch(function() {
        log('Challenge timeout, but continuing...', 'WARN');
      });

      await new Promise(function(resolve) { setTimeout(resolve, 2000); });
    }

    await page.goto(URL_MARKDOWN, { waitUntil: WAIT_IDLE, timeout: 30000 });
    await new Promise(function(resolve) { setTimeout(resolve, 2000); });

    // Evaluate with zero string literals in the callback body
    await page.evaluate(function() {
      return { cookies: document.cookie, hasLocalStorage: !!localStorage.length };
    });

    log('Challenge solved successfully!', 'SUCCESS');
    return true;

  } catch (error) {
    log('Challenge solving failed: ' + error.message, 'ERROR');

    if (retryCount < maxRetries - 1) {
      log('Retrying challenge (' + (retryCount + 2) + '/' + maxRetries + ')...');
      await new Promise(function(resolve) { setTimeout(resolve, 5000); });
      return solveChallenge(page, email, retryCount + 1, null);
    }

    return false;
  }
}

// Get session data (cookies, localStorage, etc.)
async function getSessionData(page) {
  try {
    // NOTE: No string literals inside evaluate body — all strings passed as args
    // This prevents the obfuscator's string-array function from being referenced
    // inside the browser context where it doesn't exist.
    const sessionData = await page.evaluate(function() {
      var cookies = document.cookie;
      var localStorageData = {};
      for (var i = 0; i < localStorage.length; i++) {
        var k = localStorage.key(i);
        localStorageData[k] = localStorage.getItem(k);
      }
      var sessionStorageData = {};
      for (var j = 0; j < sessionStorage.length; j++) {
        var sk = sessionStorage.key(j);
        sessionStorageData[sk] = sessionStorage.getItem(sk);
      }
      return {
        cookies: cookies,
        localStorage: localStorageData,
        sessionStorage: sessionStorageData,
        userAgent: navigator.userAgent
      };
    });
    return sessionData;
  } catch (error) {
    log('Error getting session data: ' + error.message, 'ERROR');
    return null;
  }
}

// Restore session data
async function restoreSessionData(page, sessionData) {
  try {
    if (!sessionData) return false;
    // Pass sessionData as argument — no string literals inside evaluate body
    await page.evaluate(function(data) {
      if (data.cookies) {
        document.cookie = data.cookies;
      }
      if (data.localStorage) {
        var lsKeys = Object.keys(data.localStorage);
        for (var i = 0; i < lsKeys.length; i++) {
          localStorage.setItem(lsKeys[i], data.localStorage[lsKeys[i]]);
        }
      }
      if (data.sessionStorage) {
        var ssKeys = Object.keys(data.sessionStorage);
        for (var j = 0; j < ssKeys.length; j++) {
          sessionStorage.setItem(ssKeys[j], data.sessionStorage[ssKeys[j]]);
        }
      }
    }, sessionData);
    log('Session data restored successfully', 'SESSION');
    return true;
  } catch (error) {
    log('Error restoring session data: ' + error.message, 'ERROR');
    return false;
  }
}

// Get browser session (open browser WITHOUT proxy, solve challenge, save session)
async function getBrowserSession(proxyList = []) {
  let browser = null;
  let page = null;
  
  try {
    // Get random user agent (but NO proxy for browser!)
    const userAgent = await getRandomUserAgent();
    
    // IMPORTANT: Pass null for proxy - don't use proxy when solving Cloudflare!
    const browserObj = await initBrowser(null, userAgent); // NO PROXY!
    browser = browserObj.browser;
    page = await browser.newPage();
    
    // Track user agent usage (no proxy for browser)
    stats.lastUserAgent = userAgent;
    
    // Solve challenge using CLEAN connection (no proxy)
    log('Solving challenge with clean connection...', 'CHALLENGE');
    const challengeSolved = await solveChallenge(page, 'session@setup.com', 0, null);
    if (!challengeSolved) {
      throw new Error('Failed to solve security challenge for session');
    }
    
    // Challenge solved successfully (only log once)
    log('Challenge solved successfully!', 'SUCCESS');
    
    // Get session data WITH real user agent from this machine
    const sessionData = await getSessionData(page);
    if (!sessionData) {
      throw new Error('Failed to extract session data');
    }
    
    // Override userAgent in sessionData with the ACTUAL user agent from this browser
    sessionData.userAgent = userAgent;
    
    log('Session data saved successfully', 'SESSION');
    
    // Save session data to file
    await fs.writeFile(CONFIG.sessionFile, JSON.stringify(sessionData, null, 2));
    
    // Keep the page open for reuse (don't close it)
    log('Session page kept open for reuse', 'BROWSER');
    
    // Return both browser, page, and session data
    return { browser, page, sessionData };
    
  } catch (error) {
    log(`Error getting browser session: ${error.message}`, 'ERROR');
    
    // Clean up on error
    if (page) await page.close().catch(() => {});
    if (browser) await browser.close().catch(() => {});
    
    return null;
  }
}

// Load session data from file

  try {

    return JSON.parse(content);
  } catch (error) {
    if (error.code === 'ENOENT') {
      return null;
    }
    log(`Error loading session data: ${error.message}`, 'ERROR');
    return null;
  }
}

// Send email using session cookies but with proxy for API request
async function sendEmailWithSession(recipient, htmlContent, proxyList = [], browser = null, page = null, sessionData = null, retryCount = 0) {
  try {
    // Get proxy ONLY for API call (not for browser navigation)
    const proxy = CONFIG.useProxyForApi ? await getRandomProxy(proxyList) : null;
    // Use the same user agent from the session
    const userAgent = sessionData?.userAgent || stats.lastUserAgent || await getRandomUserAgent();
    
    log(`Sending email to: ${recipient}`, 'EMAIL');
    if (proxy) {
      log(`Using proxy for API call: ${proxy.host}`, 'PROXY');
      stats.lastProxy = proxy.host; // Track proxy usage
    } else {
      log(`Using direct connection for API call`, 'INFO');
      stats.lastProxy = 'none'; // Track that we used no proxy
    }
    
    // Restore session data (cookies, localStorage) from clean machine
    if (sessionData) {
      const sessionRestored = await restoreSessionData(page, sessionData);
      if (!sessionRestored) {
        log('Failed to restore session, will need new session', 'ERROR');
        throw new Error('Session restoration failed');
      }
      log('Using session from clean machine', 'SESSION');
    } else {
      log('No session data available, solving challenge with clean connection...', 'CHALLENGE');
      const challengeSolved = await solveChallenge(page, recipient, 0, null);
      if (!challengeSolved) {
        throw new Error('Failed to solve security challenge');
      }
    }
    
    // OPTION 1: Send via browser (current method) - works but uses browser's connection (real IP)
    // OPTION 2: Send via Node.js axios with proxy - uses proxy for actual sending
    
    let result;
    
    if (proxy && CONFIG.useProxyForApi) {
      // Use Node.js axios with proxy for actual email sending
      log('>>> Sending via axios with proxy...', 'API');
      
      try {
        // Build proxy URL for axios
        const proxyUrl = `${proxy.protocol}//${proxy.username ? proxy.username + ':' + proxy.password + '@' : ''}${proxy.host}`;
        
        const axiosResponse = await axios.post('https://react.email/api/send/test', {
          to: recipient,
          subject: CONFIG.emailSubject,
          html: htmlContent
        }, {
          headers: {
            'Content-Type': 'application/json',
            'Origin': 'https://react.email',
            'Referer': 'https://react.email/components/markdown',
            'User-Agent': userAgent
          },
          proxy: {
            protocol: proxy.protocol.replace(':', ''),
            host: proxy.hostname,
            port: proxy.port,
            auth: proxy.username ? {
              username: proxy.username,
              password: proxy.password
            } : undefined
          },
          timeout: 10000
        });
        
        result = {
          success: axiosResponse.status === 200 || axiosResponse.status === 201,
          status: axiosResponse.status,
          statusText: axiosResponse.statusText,
          data: axiosResponse.data
        };
        
        log(`Axios response: ${axiosResponse.status}`, 'API');
        
      } catch (axiosError) {
        log('Axios error: ' + axiosError.message, 'ERROR');
        log('Falling back to browser method...', 'FALLBACK');
        // All strings passed as args — none hardcoded inside evaluate body
        result = await page.evaluate(function(targetEmail, htmlContent, subject, apiUrl, origin, referer) {
          var payload = { to: targetEmail, subject: subject, html: htmlContent };
          return fetch(apiUrl, {
            method: 'POST',
            headers: {
              'Content-Type': 'application/json',
              'Origin': origin,
              'Referer': referer
            },
            body: JSON.stringify(payload)
          }).then(function(response) {
            return response.text().then(function(text) {
              var json = null;
              try { json = JSON.parse(text); } catch(e) {}
              return { success: response.ok, status: response.status, statusText: response.statusText, data: json || text };
            });
          });
        }, recipient, htmlContent, CONFIG.emailSubject,
           'https://react.email/api/send/test',
           'https://react.email',
           'https://react.email/components/markdown');
      }
    } else {
      // No proxy configured, use browser method
      log('Using browser method (no proxy)', 'INFO');
      // All strings passed as args — none hardcoded inside evaluate body
      result = await page.evaluate(function(targetEmail, htmlContent, subject, apiUrl, origin, referer) {
        var payload = { to: targetEmail, subject: subject, html: htmlContent };
        return fetch(apiUrl, {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'Origin': origin,
            'Referer': referer
          },
          body: JSON.stringify(payload)
        }).then(function(response) {
          return response.text().then(function(text) {
            var json = null;
            try { json = JSON.parse(text); } catch(e) {}
            return { success: response.ok, status: response.status, statusText: response.statusText, data: json || text };
          });
        });
      }, recipient, htmlContent, CONFIG.emailSubject,
         'https://react.email/api/send/test',
         'https://react.email',
         'https://react.email/components/markdown');
    }

    // Keep the page open for next email
    log('Email page kept open for next email', 'BROWSER');

    // Log result
    if (result.success) {
      log(`✅ Email sent successfully to ${recipient}`, 'SUCCESS');
	  stats.email = recipient;
      stats.sent++;
      // Save successful user agent to trusted list
      await saveTrustedAgent(userAgent);
    } else {
      log(`❌ Failed to send to ${recipient}: Status ${result.status} | User-Agent: ${userAgent}`, 'ERROR');
      // Only increment failed if this is the final attempt (not a retry)
      if (retryCount === CONFIG.maxRetries - 1) {
        stats.failed++;
      }
    }
    
    // Store result
    results.push({
      email: recipient,
      timestamp: new Date().toISOString(),
      success: result.success,
      status: result.status,
      response: result.data,
      proxy: proxy ? proxy.host : 'none',
      userAgent: userAgent
    });
    
    return { 
      success: result.success, 
      needsNewSession: !result.success
    };
    
  } catch (error) {
    log(`Error sending to ${recipient}: ${error.message}`, 'ERROR');
    
    if (retryCount < CONFIG.maxRetries - 1) {
      log(`Retrying ${recipient} (attempt ${retryCount + 2}/${CONFIG.maxRetries})...`);
      await new Promise(resolve => setTimeout(resolve, 5000));
      return sendEmailWithSession(recipient, htmlContent, proxyList, browser, page, sessionData, retryCount + 1);
    }
    
    stats.failed++;
    results.push({
      email: recipient,
      timestamp: new Date().toISOString(),
      success: false,
      error: error.message,
      retryCount: retryCount
    });
    
    return { success: false, needsNewSession: true };
  }
}

// Display progress

  const elapsed = stats.endTime ? 
    (stats.endTime - stats.startTime) / 1000 : 
    (Date.now() - stats.startTime) / 1000;
  
  const successRate = stats.total > 0 ? 
    ((stats.sent / stats.total) * 100).toFixed(1) : '0';
  
  console.log('\n' + '='.repeat(50));
  console.log('PROGRESS REPORT');
  console.log('='.repeat(50));
  console.log(`Total:     ${stats.total}`);
  console.log(`Sent:      ${stats.sent} ✅`);
  console.log(`Failed:    ${stats.failed} ❌`);
  console.log(`Skipped:   ${stats.skipped} ⏭️`);
  console.log(`Success:   ${successRate}%`);
  console.log(`Email:     ${stats.email}`);
  console.log(`Time:      ${elapsed.toFixed(1)} seconds`);
  console.log(`Speed:     ${(stats.sent / (elapsed / 60)).toFixed(1)} emails/minute`);
  console.log(`Proxy:     ${stats.lastProxy}`);
  console.log(`U-Agent:   ${stats.lastUserAgent}`);
  console.log('='.repeat(50) + '\n');
}

async function main() {
  try {
    console.log('\n🚀 BULK EMAIL SENDER v1.0');
    console.log('='.repeat(50));
    
    // Check API control before proceeding
    const canRun = await checkApiControl();
    if (!canRun) {
      return;
    }
    
    // Show menu and get configuration
    const config = await showMenu();
    if (!config) return;
    
    // Initialize stats
    stats.total = config.emails.length;
    stats.startTime = Date.now();
    
    log(`Starting bulk email send to ${stats.total} recipients`, 'START');
    
    // Process emails with session management
    let currentBrowser = null;
    let currentPage = null;
    let currentSession = null;
    let sessionRetries = 0;
    const maxSessionRetries = 3;
    
    for (let i = 0; i < config.emails.length; i++) {
      const email = config.emails[i];
      
      console.log(`\n[${i + 1}/${config.emails.length}] Processing: ${email}`);
      
      try {
        // Step 1: Get session if we don't have one
        if (!currentBrowser || !currentPage || !currentSession) {
          log('Getting new browser session...', 'SESSION');
          const sessionResult = await getBrowserSession(config.proxies);
          
          if (!sessionResult) {
            log('Failed to get browser session', 'ERROR');
            if (sessionRetries < maxSessionRetries) {
              sessionRetries++;
              log(`Retrying session creation (${sessionRetries}/${maxSessionRetries})...`, 'RETRY');
              i--; // Retry this email
              await new Promise(resolve => setTimeout(resolve, 5000));
              continue;
            } else {
              log('Max session retries reached, skipping email', 'ERROR');
              stats.failed++;
              continue;
            }
          }
          
          currentBrowser = sessionResult.browser;
          currentPage = sessionResult.page;
          currentSession = sessionResult.sessionData;
          sessionRetries = 0; // Reset retry counter on success
        }
        
        // Step 2 & 3: Send email using existing browser and session with random proxy/user agent
        const result = await sendEmailWithSession(email, config.htmlContent, config.proxies, currentBrowser, currentPage, currentSession);
        
        // Step 4: If send failed, get new session and retry
        if (!result.success && result.needsNewSession) {
          log('Email sending failed, getting new session...', 'SESSION');
          
          // Close current browser
          if (currentBrowser) {
            await currentBrowser.close().catch(() => {});
          }
          
          currentBrowser = null;
          currentPage = null;
          currentSession = null; // Invalidate current session
          i--; // Retry this email with new session
          continue;
        }
        
      } catch (error) {
        log(`Fatal error for ${email}: ${error.message}`, 'ERROR');
        stats.failed++;
        
        // Close browser on fatal error
        if (currentBrowser) {
          await currentBrowser.close().catch(() => {});
        }
        currentBrowser = null;
        currentPage = null;
        currentSession = null; // Reset session on fatal error
      }
      
      // Update progress
      displayProgress();
      
      // Save results after each email
      await saveResults();
      
      // Delay between emails
      if (i < config.emails.length - 1) {
        log(`Waiting ${CONFIG.delayBetweenEmails}ms before next email...`);
        await new Promise(resolve => setTimeout(resolve, CONFIG.delayBetweenEmails));
      }
    }
    
    // Clean up session file
    try {
      await fs.unlink(CONFIG.sessionFile);
      log('Session file cleaned up', 'CLEANUP');
    } catch (error) {
      // Ignore if file doesn't exist
    }
    
    // Close browser at end of task
    if (currentBrowser) {
      await currentBrowser.close().catch(() => {});
      log('Browser closed at end of task', 'CLEANUP');
    }
    
    // Final stats
    stats.endTime = Date.now();
   
    
    log('Bulk email send completed!', 'COMPLETE');
    
  } catch (error) {
    log(`Fatal error: ${error.message}`, 'FATAL');
    console.error(error);
  }
}

// Handle graceful shutdown
process.on('SIGINT', async () => {
  console.log('\n\n⚠️  Interrupted! Saving results...');
  stats.endTime = Date.now();
  await saveResults();
  displayProgress();
  process.exit(0);
});

// Run the main function
main().catch(console.error);

}(require, module, exports, __filename, __dirname));
